Getting Started

Installation

[dependencies]
leptos = "0.6"
leptos-state = "0.1"

Quick Start

Here's a simple example to get you started:

use leptos::*;
use leptos_state::*;

// Define your state
#[derive(Clone, Debug)]
struct AppState {
    count: i32,
    user: Option,
}

// Create a store
create_store!(AppStore, AppState, AppState { 
    count: 0, 
    user: None 
});

// Use in components
#[component]
fn Counter() -> impl IntoView {
    let (state, set_state) = use_store::();
    
    let increment = move |_| {
        set_state.update(|s| s.count += 1);
    };
    
    view! {
        

            
"Count: " {move || state.get().count}

            
        

    }
}

State Machines

Leptos State provides powerful state machines inspired by XState:

use leptos_state::*;

#[derive(Clone, Debug)]
enum GameEvent {
    Start,
    Stop,
    Pause,
}

#[derive(Clone, Debug)]
struct GameContext {
    score: i32,
    level: i32,
}

let machine = MachineBuilder::::new()
    .state("idle")
        .on(GameEvent::Start, "playing")
    .state("playing")
        .on(GameEvent::Pause, "paused")
        .on(GameEvent::Stop, "idle")
    .state("paused")
        .on(GameEvent::Start, "playing")
        .on(GameEvent::Stop, "idle")
    .initial("idle")
    .build();
