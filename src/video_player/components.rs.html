15#[component]
16pub fn ControlsOverlay(
17    store: crate::state_machine::VideoPlayerStore,
18    on_play: impl Fn(MouseEvent) + 'static + Copy,
19    on_pause: impl Fn(MouseEvent) + 'static + Copy,
20    _on_toggle: impl Fn(MouseEvent) + 'static + Copy,
21    on_seek: impl Fn(f64) + 'static + Copy,
22    on_volume_change: impl Fn(f64) + 'static + Copy,
23    on_volume_mute: impl Fn(MouseEvent) + 'static + Copy,
24    on_fullscreen: impl Fn(MouseEvent) + 'static + Copy,
25) -> impl IntoView {
26    let is_playing = move || store.state.get() == crate::state_machine::VideoPlayerState::Playing;
27    let current_time = move || store.context.get().video_current_time;
28    let duration = move || store.context.get().video_duration.unwrap_or(1.0);
29    let progress = move || {
30        let current = current_time();
31        let dur = duration();
32        if dur > 0.0 {
33            current / dur
34        } else {
35            0.0
36        }
37    };
38
39    view! {
40        <div class="controls-overlay">
41            <div class="controls-top">
42                <div class="video-title">"Big Buck Bunny"</div>
43            </div>
44
45            <div class="controls-center">
46                <button
47                    class="play-button"
48                    on:click=move |ev| if is_playing() { on_pause(ev) } else { on_play(ev) }
49                >
50                    {if is_playing() { "⏸️" } else { "▶️" }}
51                </button>
52            </div>
53
54            <div class="controls-bottom">
55                <div class="progress-container">
56                    <div class="progress-bar">
57                        <div
58                            class="progress-fill"
59                            style=move || format!("width: {}%", progress() * 100.0)
60                        ></div>
61                        <div
62                            class="progress-handle"
63                            style=move || format!("left: {}%", progress() * 100.0)
64                            on:mousedown=move |_| {
65                                // Handle seeking
66                                on_seek(progress());
67                            }
68                        ></div>
69                    </div>
70                </div>
71
72                <div class="time-display">
73                    <span>{move || format_time(current_time())}</span>
74                    " / "
75                    <span>{move || format_time(duration())}</span>
76                </div>
77
78                <div class="volume-controls">
79                    <button
80                        class="volume-button"
81                        on:click=on_volume_mute
82                    >
83                        {move || if store.context.get().muted { "🔇" } else { "🔊" }}
84                    </button>
85                    <input
86                        type="range"
87                        class="volume-slider"
88                        min="0"
89                        max="1"
90                        step="0.1"
91                        value=move || store.context.get().volume
92                        on:input=move |ev| {
93                            let value = event_target_value(&ev).parse::<f64>().unwrap_or(1.0);
94                            on_volume_change(value);
95                        }
96                    />
97                </div>
98
99                <button
100                    class="fullscreen-button"
101                    on:click=on_fullscreen
102                >
103                    "⛶"
104                </button>
105            </div>
106        </div>
107    }
108}

133fn format_time(seconds: f64) -> String {
134    let minutes = (seconds / 60.0) as u32;
135    let seconds = (seconds % 60.0) as u32;
136    format!("{:02}:{:02}", minutes, seconds)
137}
