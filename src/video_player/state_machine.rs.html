3/// Context for the video player state machine
4#[derive(Debug, Clone, PartialEq)]
5pub struct VideoPlayerContext {
6    pub video_src: String,
7    pub video_poster: String,
8    pub current_video_src: Option<String>,
9    pub video_duration: Option<f64>,
10    pub video_current_time: f64,
11    pub volume: f64,
12    pub muted: bool,
13    pub animation_action_timestamp: String,
14    pub is_touch_device: bool,
15}
16
17impl Default for VideoPlayerContext {
18    fn default() -> Self {
19        Self {
20            video_src: String::new(),
21            video_poster: String::new(),
22            current_video_src: None,
23            video_duration: None,
24            video_current_time: 0.0,
25            volume: 1.0,
26            muted: false,
27            animation_action_timestamp: String::new(),
28            is_touch_device: false,
29        }
30    }
31}

33/// Video player state using Leptos signals
34#[derive(Debug, Clone, PartialEq)]
35pub enum VideoPlayerState {
36    Stopped,
37    Loading,
38    Playing,
39    Paused,
40    #[allow(dead_code)]
41    Ended,
42}
43
44#[derive(Debug, Clone, PartialEq)]
45pub enum AnimationType {
46    Playing,
47    Paused,
48    Backward,
49    Forward,
50}
51
52/// Video player store using Leptos signals
53#[derive(Clone, Copy)]
54pub struct VideoPlayerStore {
55    pub state: ReadSignal<VideoPlayerState>,
56    pub set_state: WriteSignal<VideoPlayerState>,
57    pub context: ReadSignal<VideoPlayerContext>,
58    pub set_context: WriteSignal<VideoPlayerContext>,
59    pub is_loading: ReadSignal<bool>,
60    pub set_loading: WriteSignal<bool>,
61    pub show_controls: ReadSignal<bool>,
62    pub set_show_controls: WriteSignal<bool>,
63    pub animation: ReadSignal<Option<AnimationType>>,
64    pub set_animation: WriteSignal<Option<AnimationType>>,
65    pub is_fullscreen: ReadSignal<bool>,
66    pub set_fullscreen: WriteSignal<bool>,
67}
68
69impl VideoPlayerStore {
70    pub fn new(initial_context: VideoPlayerContext) -> Self {
71        let (state, set_state) = signal(VideoPlayerState::Stopped);
72        let (context, set_context) = signal(initial_context);
73        let (is_loading, set_loading) = signal(false);
74        let (show_controls, set_show_controls) = signal(false);
75        let (animation, set_animation) = signal(None);
76        let (is_fullscreen, set_fullscreen) = signal(false);
77
78        Self {
79            state,
80            set_state,
81            context,
82            set_context,
83            is_loading,
84            set_loading,
85            show_controls,
86            set_show_controls,
87            animation,
88            set_animation,
89            is_fullscreen,
90            set_fullscreen,
91        }
92    }
93
94    pub fn play(&self) {
95        self.set_state.set(VideoPlayerState::Playing);
96        self.set_loading.set(false);
97    }
98
99    pub fn pause(&self) {
100        self.set_state.set(VideoPlayerState::Paused);
101    }
102
103    pub fn stop(&self) {
104        self.set_state.set(VideoPlayerState::Stopped);
105    }
106
107    pub fn set_loading(&self, loading: bool) {
108        self.set_loading.set(loading);
109        if loading {
110            self.set_state.set(VideoPlayerState::Loading);
111        }
112    }
113
114    pub fn update_time(&self, current_time: f64) {
115        self.set_context.update(|ctx| {
116            ctx.video_current_time = current_time;
117        });
118    }
119
120    pub fn update_duration(&self, duration: f64) {
121        self.set_context.update(|ctx| {
122            ctx.video_duration = Some(duration);
123        });
124    }
125
126    pub fn set_volume(&self, volume: f64) {
127        self.set_context.update(|ctx| {
128            ctx.volume = volume;
129        });
130    }
131
132    pub fn toggle_mute(&self) {
133        self.set_context.update(|ctx| {
134            ctx.muted = !ctx.muted;
135        });
136    }
137
138    pub fn animate(&self, animation: AnimationType) {
139        self.set_animation.set(Some(animation));
140        // Clear animation after a delay
141        set_timeout(
142            move || {
143                // Animation will be cleared by the component
144            },
145            std::time::Duration::from_millis(600),
146        );
147    }
148
149    pub fn toggle_fullscreen(&self) {
150        self.set_fullscreen.set(!self.is_fullscreen.get());
151    }
152}
