1use leptos::ev::SubmitEvent;
2use leptos::prelude::*;
3// use leptos::*;
4use std::collections::HashMap;
5use uuid::Uuid;
6
7#[derive(Debug, Clone, PartialEq)]
8pub struct Todo {
9    pub id: Uuid,
10    pub title: String,
11    pub completed: bool,
12}
13
14impl Todo {
15    pub fn new(title: String) -> Self {
16        Self {
17            id: Uuid::new_v4(),
18            title,
19            completed: false,
20        }
21    }
22}
23
24#[component]
25pub fn SimpleTodoApp() -> impl IntoView {
26    let (todos, set_todos) = signal(HashMap::new());
27    let (new_todo_title, set_new_todo_title) = signal(String::new());
28
29    let handle_add_todo = move |ev: SubmitEvent| {
30        ev.prevent_default();
31        let title = new_todo_title.get();
32        if !title.trim().is_empty() {
33            let todo = Todo::new(title);
34            set_todos.update(|todos| {
35                todos.insert(todo.id, todo);
36            });
37            set_new_todo_title.set(String::new());
38        }
39    };
40
41    view! {
42        <div style="max-width: 600px; margin: 0 auto; padding: 2rem;">
43            <h1>"Simple Todo App"</h1>
44            <p>"Built with Leptos and Rust"</p>
45
46            <form on:submit=handle_add_todo>
47                <input
48                    type="text"
49                    placeholder="What needs to be done?"
50                    prop:value=new_todo_title
51                    on:input=move |ev| {
52                        let value = ev.target().unwrap().value_of().as_string().unwrap_or_default();
53                        set_new_todo_title.set(value);
54                    }
55                    required=true
56                    style="width: 100%; padding: 0.5rem; margin-bottom: 1rem;"
57                />
58                <button type="submit" style="padding: 0.5rem 1rem; background: #3b82f6; color: white; border: none; border-radius: 0.25rem;">
59                    "Add Todo"
60                </button>
61            </form>
62
63            <div>
64                <p>"Todo List:"</p>
65                {move || {
66                    let todo_list: Vec<Todo> = todos.get().values().cloned().collect();
67                    view! {
68                        <div>
69                            {todo_list.into_iter().map(|todo| {
70                                let todo_id = todo.id;
71                                let completed = todo.completed;
72                                let title = todo.title.clone();
73
74                                view! {
75                                    <div style="display: flex; align-items: center; padding: 0.5rem; border: 1px solid #e2e8f0; margin-bottom: 0.5rem; border-radius: 0.25rem;">
76                                        <input
77                                            type="checkbox"
78                                            checked=completed
79                                            on:change=move |_| {
80                                                set_todos.update(|todos| {
81                                                    if let Some(todo) = todos.get_mut(&todo_id) {
82                                                        todo.completed = !todo.completed;
83                                                    }
84                                                });
85                                            }
86                                            style="margin-right: 0.5rem;"
87                                        />
88                                        <span style=move || if completed { "text-decoration: line-through; color: #64748b; flex: 1;" } else { "flex: 1;" }>
89                                            {title}
90                                        </span>
91                                        <button
92                                            on:click=move |_| {
93                                                set_todos.update(|todos| {
94                                                    todos.remove(&todo_id);
95                                                });
96                                            }
97                                            style="background: #ef4444; color: white; border: none; padding: 0.25rem 0.5rem; border-radius: 0.25rem; cursor: pointer;"
98                                        >
99                                            "Delete"
100                                        </button>
101                                    </div>
102                                }
103                            }).collect::<Vec<_>>()}
104                        </div>
105                    }.into_view()
106                }}
107            </div>
108        </div>
109    }
110}
