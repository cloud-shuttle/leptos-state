7// Data Models
8#[derive(Debug, Clone, PartialEq)]
9pub struct Metric {
10    pub id: Uuid,
11    pub name: String,
12    pub value: f64,
13    pub change: f64,
14    pub trend: Trend,
15    pub category: MetricCategory,
16    pub timestamp: DateTime<Utc>,
17}
18
19#[derive(Debug, Clone, PartialEq)]
20#[allow(dead_code)]
21pub enum Trend {
22    Up,
23    Down,
24    Stable,
25}
26
27#[derive(Debug, Clone, PartialEq)]
28pub enum MetricCategory {
29    Revenue,
30    Users,
31    Performance,
32    Engagement,
33}
