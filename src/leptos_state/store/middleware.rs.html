4/// Trait for store middleware
5pub trait Middleware<S: Store> {
6    /// Wrap the next middleware/store operation
7    fn wrap(&self, next: Box<dyn Fn(&S::State) -> S::State>) -> Box<dyn Fn(&S::State) -> S::State>;
8}
9
10/// Chain multiple middleware together
11pub struct MiddlewareChain<S: Store> {
12    middlewares: Vec<Box<dyn Middleware<S>>>,
13}
14
15impl<S: Store> MiddlewareChain<S> {
16    pub fn new() -> Self {
17        Self {
18            middlewares: Vec::new(),
19        }
20    }
21
22    pub fn add(mut self, middleware: Box<dyn Middleware<S>>) -> Self {
23        self.middlewares.push(middleware);
24        self
25    }
26
27    pub fn apply(
28        &self,
29        base: impl Fn(&S::State) -> S::State + 'static,
30    ) -> impl Fn(&S::State) -> S::State {
31        let mut result: Box<dyn Fn(&S::State) -> S::State> = Box::new(base);
32
33        for middleware in self.middlewares.iter().rev() {
34            result = middleware.wrap(result);
35        }
36
37        move |state| result(state)
38    }
39}
40
41impl<S: Store> Default for MiddlewareChain<S> {
42    fn default() -> Self {
43        Self::new()
44    }
45}

47/// Logger middleware for debugging state changes
48pub struct LoggerMiddleware<S: Store> {
49    prefix: String,
50    _phantom: PhantomData<S>,
51}
52
53impl<S: Store> LoggerMiddleware<S> {
54    pub fn new(prefix: impl Into<String>) -> Self {
55        Self {
56            prefix: prefix.into(),
57            _phantom: PhantomData,
58        }
59    }
60}
61
62impl<S: Store> Middleware<S> for LoggerMiddleware<S>
63where
64    S::State: std::fmt::Debug,
65{
66    fn wrap(&self, next: Box<dyn Fn(&S::State) -> S::State>) -> Box<dyn Fn(&S::State) -> S::State> {
67        let prefix = self.prefix.clone();
68        Box::new(move |state| {
69            tracing::debug!("{}: Previous state: {:?}", prefix, state);
70            let new_state = next(state);
71            tracing::debug!("{}: New state: {:?}", prefix, new_state);
72            new_state
73        })
74    }
75}

90        }
91    }

94#[cfg(feature = "persist")]
95impl<S: Store> Middleware<S> for PersistMiddleware<S>
96where
97    S::State: serde::Serialize + for<'de> serde::Deserialize<'de>,
98{
99    fn wrap(&self, next: Box<dyn Fn(&S::State) -> S::State>) -> Box<dyn Fn(&S::State) -> S::State> {
100        let key = self.key.clone();
101        Box::new(move |state| {
102            let new_state = next(state);
103
104            // Save to localStorage using the utility function
105            if let Err(e) = crate::store::save_to_storage(&key, &new_state) {
106                tracing::warn!("Failed to persist state: {:?}", e);
107            }
108
109            new_state
110        })
111    }
112}

114/// Validation middleware to ensure state integrity
115pub struct ValidationMiddleware<S: Store, F> {
116    validator: F,
117    _phantom: PhantomData<S>,
118}
119
120impl<S: Store, F> ValidationMiddleware<S, F>
121where
122    F: Fn(&S::State) -> bool,
123{
124    pub fn new(validator: F) -> Self {
125        Self {
126            validator,
127            _phantom: PhantomData,
128        }
129    }
130}
131
132impl<S: Store, F> Middleware<S> for ValidationMiddleware<S, F>
133where
134    F: Fn(&S::State) -> bool + Clone + 'static,
135{
136    fn wrap(&self, next: Box<dyn Fn(&S::State) -> S::State>) -> Box<dyn Fn(&S::State) -> S::State> {
137        let validator = self.validator.clone();
138        Box::new(move |state| {
139            let new_state = next(state);
140
141            if validator(&new_state) {
142                new_state
143            } else {
144                tracing::warn!("State validation failed, keeping previous state");
145                state.clone()
146            }
147        })
148    }
149}

167            .add(Box::new(LoggerMiddleware::new("second")));
168
169        // Test that the chain can be created and applied
170        let wrapped = chain.apply(|state| TestState {
171            count: state.count + 1,
172        });
173        let result = wrapped(&TestState { count: 0 });
174        assert_eq!(result.count, 1);
175    }

177    #[test]

179        let validator: ValidationMiddleware<TestStore, _> =
180            ValidationMiddleware::new(|state: &TestState| state.count >= 0);
181
182        let wrapped = validator.wrap(Box::new(|_| TestState { count: -1 }));
183        let result: TestState = wrapped(&TestState { count: 5 });
184
185        // Should keep the original state since -1 is invalid
