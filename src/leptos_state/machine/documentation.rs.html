155    /// Generate documentation for the state machine
156    pub fn generate_documentation(&self) -> StateResult<Vec<GeneratedDocument>> {
157        let start_time = Instant::now();
158        let mut generated_docs = Vec::new();
159
160        // Generate documentation for each format
161        for format in &self.config.output_formats {
162            let doc = self.generate_format_documentation(format)?;
163            generated_docs.push(doc);
164        }
165
166        // Save documentation to files
167        if !self.config.output_directory.is_empty() {
168            self.save_documentation(&generated_docs)?;
169        }
170
171        // Update generated docs
172        if let Ok(mut docs) = self.generated_docs.write() {
173            *docs = generated_docs.clone();
174        }
175
176        println!("Documentation generated in {:?}", start_time.elapsed());
177        Ok(generated_docs)
178    }

180    /// Generate documentation for a specific format
181    fn generate_format_documentation(
182        &self,
183        format: &DocumentationFormat,
184    ) -> StateResult<GeneratedDocument> {
185        let content = match format {
186            DocumentationFormat::Markdown => self.generate_markdown_documentation()?,
187            DocumentationFormat::Html => self.generate_html_documentation()?,
188            DocumentationFormat::Pdf => self.generate_pdf_documentation()?,
189            DocumentationFormat::AsciiDoc => self.generate_asciidoc_documentation()?,
190            DocumentationFormat::Rst => self.generate_rst_documentation()?,
191            DocumentationFormat::Json => self.generate_json_documentation()?,
192            DocumentationFormat::Yaml => self.generate_yaml_documentation()?,
193            DocumentationFormat::Custom(ref custom_format) => {
194                self.generate_custom_documentation(custom_format)?
195            }
196        };
197
198        Ok(GeneratedDocument {
199            format: format.clone(),
200            content,
201            generated_at: Instant::now(),
202            file_path: self.get_output_path(format),
203        })
204    }

602        let filename = match format {

616    /// Save documentation to files
617    fn save_documentation(&self, docs: &[GeneratedDocument]) -> StateResult<()> {
618        // Create output directory if it doesn't exist
619        if let Err(_) = fs::create_dir_all(&self.config.output_directory) {
620            return Err(StateError::custom(format!(
621                "Failed to create output directory: {}",
622                self.config.output_directory
623            )));
624        }
625
626        // Save each document
627        for doc in docs {
628            if let Err(e) = fs::write(&doc.file_path, &doc.content) {
629                return Err(StateError::custom(format!(
630                    "Failed to write documentation to {}: {}",
631                    doc.file_path, e
632                )));
633            }
634            println!("Documentation saved to: {}", doc.file_path);
635        }
636
637        Ok(())
638    }

679        ));
680        index.push_str(&format!(
681            "- Output directory: {}\n",
682            self.config.output_directory
683        ));
684        index.push_str(&format!(
685            "- Formats: {}\n",
686            self.config
687                .output_formats
688                .iter()
689                .map(|f| format!("{:?}", f))
690                .collect::<Vec<_>>()
691                .join(", ")
692        ));
693
694        Ok(index)
695    }
696}

745/// Extension trait for adding documentation to machines
746pub trait MachineDocumentationExt<C: Send + Sync, E> {
747    /// Add documentation generation capabilities to the machine
748    fn with_documentation(self, config: DocumentationConfig) -> DocumentationGenerator<C, E>;
749}
750
751impl<C, E> MachineDocumentationExt<C, E> for Machine<C, E>
752where
753    C: Clone + std::fmt::Debug + Send + Sync,
754    E: Clone + std::fmt::Debug + Event + Send + Sync,
755{
756    fn with_documentation(self, config: DocumentationConfig) -> DocumentationGenerator<C, E> {
757        DocumentationGenerator::new(self, config)
758    }

761/// Documentation builder for fluent configuration
762pub struct DocumentationBuilder<C: Send + Sync, E> {
763    machine: Machine<C, E>,
764    config: DocumentationConfig,
765}
766
767impl<C, E> DocumentationBuilder<C, E>
768where
769    C: Clone + std::fmt::Debug + Send + Sync,
770    E: Clone + std::fmt::Debug + Event + Send + Sync,
771{
772    pub fn new(machine: Machine<C, E>) -> Self {
773        Self {
774            machine,
775            config: DocumentationConfig::default(),
776        }
777    }
778
779    pub fn with_config(mut self, config: DocumentationConfig) -> Self {
780        self.config = config;
781        self
782    }
783
784    pub fn with_format(mut self, format: DocumentationFormat) -> Self {
785        self.config.output_formats.push(format);
786        self
787    }
788
789    pub fn with_formats(mut self, formats: Vec<DocumentationFormat>) -> Self {
790        self.config.output_formats = formats;
791        self
792    }
793
794    pub fn with_output_directory(mut self, directory: String) -> Self {
795        self.config.output_directory = directory;
796        self
797    }
798
799    pub fn with_template(mut self, template: DocumentationTemplate) -> Self {
800        self.config.template = template;
801        self
802    }
803
804    pub fn with_diagrams(mut self, include: bool) -> Self {
805        self.config.include_diagrams = include;
806        self
807    }
808
809    pub fn with_code_examples(mut self, include: bool) -> Self {
810        self.config.include_code_examples = include;
811        self
812    }
813
814    pub fn with_api_docs(mut self, include: bool) -> Self {
815        self.config.include_api_docs = include;
816        self
817    }
818
819    pub fn with_usage_examples(mut self, include: bool) -> Self {
820        self.config.include_usage_examples = include;
821        self
822    }
823
824    pub fn with_metadata(mut self, key: String, value: String) -> Self {
825        self.config.metadata.insert(key, value);
826        self
827    }
828
829    pub fn with_styling(mut self, styling: DocumentationStyling) -> Self {
830        self.config.styling = styling;
831        self
832    }
833
834    pub fn build(self) -> DocumentationGenerator<C, E> {
835        DocumentationGenerator::new(self.machine, self.config)
836    }
